# Runtime dependencies. This file is what SAM installs into the Lambda package, so
# keep it to what production actually needs — anything here ships to AWS and adds
# to cold-start time.
#
# boto3 is deliberately absent: the Lambda runtime already provides it. Install it
# locally (pip install -r requirements-dev.txt) if you exercise the DynamoDB path
# on your machine.
requests>=2.31

# Mints Google access tokens from the service-account key. Tokens live one
# hour, so a pasted token cannot work in production.
google-auth>=2.28

# Draws the campaign report's bar charts to PNG. python-bidi reorders mixed
# Hebrew/Latin chart labels for correct display.
Pillow>=10.2
python-bidi>=0.4

# Renders the campaign report PDF via headless Chromium — full-bleed banner, zero
# margins, crisp output that the Drive Docs path could not produce. On Lambda the
# Chromium binary comes from a layer, not Playwright's download: set
# PLAYWRIGHT_CHROMIUM_PATH to the layer binary (see TASKS.md). The contract PDF
# still uses the Drive path and needs none of this.
playwright>=1.40
